import sys
from PySide6.QtCore import Qt, QTimer, QPropertyAnimation
from PySide6.QtGui import QPixmap
from PySide6.QtWidgets import QApplication, QLabel, QWidget, QVBoxLayout


class SplashScreen(QWidget):
    def __init__(self, program_name="IdocInTheBox Tools", version="v3.5.0"):
        super().__init__()

        # --- WINDOW CONFIG ---
        self.setWindowFlags(Qt.FramelessWindowHint | Qt.WindowStaysOnTopHint)
        self.setAttribute(Qt.WA_TranslucentBackground)

        # Base window size (adjust if needed)
        self.setFixedSize(900, 500)

        # --- BACKGROUND IMAGE ---
        self.bg_label = QLabel(self)
        pix = QPixmap("idocinthebox_creations.jpg")

        if pix.isNull():
            print("ERROR: Cannot load 'idocinthebox_creations.jpg'")

        scaled_pix = pix.scaled(
            self.size(),
            Qt.KeepAspectRatioByExpanding,
            Qt.SmoothTransformation
        )

        self.bg_label.setPixmap(scaled_pix)
        self.bg_label.setGeometry(0, 0, self.width(), self.height())
        self.bg_label.lower()   # Ensures background stays *behind* all text

        # --- TEXT OVERLAY LAYOUT ---
        layout = QVBoxLayout(self)
        layout.setContentsMargins(0, 0, 0, 0)

        # Title text
        self.title = QLabel(program_name)
        self.title.setStyleSheet("""
            color: white;
            font-size: 40px;
            font-weight: bold;
            text-shadow: 2px 2px 4px #000000;
        """)
        self.title.setAlignment(Qt.AlignCenter)

        # Version text
        self.version = QLabel(version)
        self.version.setStyleSheet("""
            color: #DDDDDD;
            font-size: 22px;
            text-shadow: 1px 1px 4px #000;
        """)
        self.version.setAlignment(Qt.AlignCenter)

        # Animated Loading Text
        self.loading = QLabel("Loading")
        self.loading.setStyleSheet("""
            color: white;
            font-size: 18px;
            text-shadow: 1px 1px 4px #000;
        """)
        self.loading.setAlignment(Qt.AlignCenter)

        layout.addStretch()
        layout.addWidget(self.title)
        layout.addWidget(self.version)
        layout.addSpacing(50)
        layout.addWidget(self.loading)
        layout.addStretch()

        # --- LOADING DOTS ANIMATION ---
        self.dot_state = 0
        self.dot_timer = QTimer()
        self.dot_timer.timeout.connect(self.animate_loading_text)
        self.dot_timer.start(450)

        # --- FADE-IN ANIMATION ---
        self.setWindowOpacity(0)
        self.fade_in = QPropertyAnimation(self, b"windowOpacity")
        self.fade_in.setDuration(1200)
        self.fade_in.setStartValue(0)
        self.fade_in.setEndValue(1)
        self.fade_in.start()

    def animate_loading_text(self):
        dots = ["", ".", "..", "..."]
        self.dot_state = (self.dot_state + 1) % 4
        self.loading.setText(f"Loading{dots[self.dot_state]}")


# --- DEMO MAIN WINDOW ---
class MainWindow(QWidget):
    def __init__(self):
        super().__init__()
        self.setWindowTitle("Main Application Window")
        self.resize(900, 600)


# --- ENTRY POINT ---
if __name__ == "__main__":
    app = QApplication(sys.argv)

    # ★ IMPORTANT ★  
    # Show the splash screen IMMEDIATELY upon double-clicking the app
    splash = SplashScreen()
    splash.show()
    app.processEvents()   # <-- This makes the splash appear instantly with zero delay

    # Simulate heavy loading or initialization without blocking UI
    def start_main():
        splash.close()
        main = MainWindow()
        main.show()

    QTimer.singleShot(3000, start_main)   # simulate initialization load

    sys.exit(app.exec())
